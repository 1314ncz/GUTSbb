local function updateCharacterRefs()

                if trimmed ~= "" then
                    huntList[trimmed] = true

            end
        end
    end)
end

-- ========== Player List Management ==========
local function updatePlayerList()
    local playersList = players:GetPlayers()
    lastPlayerList = {}
    for i, p in ipairs(playersList) do
        if p ~= player then
            table.insert(lastPlayerList, {
                name = p.Name,
                displayName = p.DisplayName,
                player = p
            })
        end
    end
    return lastPlayerList
end

local function getPlayerByIndex(index)
    updatePlayerList()
    if index < 1 or index > #lastPlayerList then return nil end
    return lastPlayerList[index].player
end

local function findPlayer(input)
    if not input or input == "" then return nil end
    
    local index = tonumber(input)
    if index then
        local p = getPlayerByIndex(index)
        if p then return p end
        return nil
    end
    
    local playersList = {}
    for _, p in ipairs(players:GetPlayers()) do
        if p ~= player then
            table.insert(playersList, p)
        end
    end
    
    local inputLower = input:lower()
    
    for _, p in ipairs(playersList) do
        if p.DisplayName:lower() == inputLower then
            return p
        end
    end
    
    for _, p in ipairs(playersList) do
        if p.Name:lower() == inputLower then
            return p
        end
    end
    
    local matches = {}
    for _, p in ipairs(playersList) do
        if p.DisplayName:lower():find(inputLower, 1, true) then
            table.insert(matches, p)
        end
    end
    
    if #matches == 0 then
        for _, p in ipairs(playersList) do
            if p.Name:lower():find(inputLower, 1, true) then
                table.insert(matches, p)
            end
        end
    end
    
    if #matches == 1 then
        return matches[1]
    elseif #matches > 1 then
        local matchNames = {}
        for _, p in ipairs(matches) do
            table.insert(matchNames, p.DisplayName)
        end
        output("Multiple matches found: " .. table.concat(matchNames, ", ") .. ", please use a more precise name", false)
        return nil
    end
    
    return nil
end

local function refreshPlayerList()
    updatePlayerList()
    local names = {}
    for i, info in ipairs(lastPlayerList) do
        table.insert(names, i .. ":" .. info.displayName .. "(@" .. info.name .. ")")
    end
    if #names > 0 then
        output("Online players: " .. table.concat(names, ", "), true)
    else
        output("No other players online", true)
    end
end

local function sendChatMessage(message, channel)
    channel = channel or "All"
    local TextChatService = game:GetService("TextChatService")
    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local textChannel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if textChannel then
            pcall(function() textChannel:SendAsync(message) end)
        end
    else
        local sayRequest = game:GetService("ReplicatedStorage"):FindFirstChild("DefaultChatSystemChatEvents")
        if sayRequest then
            sayRequest = sayRequest:FindFirstChild("SayMessageRequest")
            if sayRequest then
                pcall(function() sayRequest:FireServer(message, channel) end)
            end
        end
    end
end

-- ========== Auto-equip Weapon ==========
local function autoEquipWeapon()
    local playerGui = player:WaitForChild("PlayerGui")
    
    local function clickInventoryTab()
        for _, gui in ipairs(playerGui:GetChildren()) do
            if gui:IsA("ScreenGui") then
                for _, btn in ipairs(gui:GetDescendants()) do
                    if (btn:IsA("ImageButton") or btn:IsA("TextButton")) and btn.Visible then
                        local btnText = btn.Text or btn.Name or ""
                        if btnText == "库存" or btnText == "Inventory" or btnText:find("库存") then
                            btn:Click()
                            output("✅ Clicked 'Inventory' tab", true)
                            task.wait(0.5)
                            return true
                        end
                    end
                end
            end
        end
        output("❌ 'Inventory' tab not found", false)
        return false
    end
    
    local function equipWeapon(weaponName)
        weaponName = weaponName or "Katana"
        
        for _, gui in ipairs(playerGui:GetChildren()) do
            if gui:IsA("ScreenGui") then
                for _, btn in ipairs(gui:GetDescendants()) do
                    if (btn:IsA("ImageButton") or btn:IsA("TextButton")) and btn.Visible then
                        local btnText = btn.Text or btn.Name or ""
                        if btnText == weaponName or btnText:find(weaponName) then
                            btn:Click()
                            output("✅ Equipped weapon: " .. weaponName, true)
                            task.wait(0.3)
                            return true
                        end
                    end
                end
            end
        end
        
        local weapons = {"Katana", "Dagger", "Axe", "Baseball Bat", "Soda Can", "Rusty Pipe", "Broken Bottle"}
        for _, gui in ipairs(playerGui:GetChildren()) do
            if gui:IsA("ScreenGui") then
                for _, btn in ipairs(gui:GetDescendants()) do
                    if (btn:IsA("ImageButton") or btn:IsA("TextButton")) and btn.Visible then
                        local btnText = btn.Text or btn.Name or ""
                        for _, wName in ipairs(weapons) do
                            if btnText == wName or btnText:find(wName) then
                                btn:Click()
                                output("✅ Equipped weapon: " .. wName, true)
                                task.wait(0.3)
                                return true
                            end
                        end
                    end
                end
            end
        end
        return false
    end
    
    output("🔧 Starting auto-equip weapon...", true)
    
    if clickInventoryTab() then
        task.wait(0.5)
    else
        output("Trying to press I to open backpack...", true)
        local VirtualInputManager = game:GetService("VirtualInputManager")
        VirtualInputManager:SendKeyEvent(true, "I", false, game)
        task.wait(0.1)
        VirtualInputManager:SendKeyEvent(false, "I", false, game)
        task.wait(0.5)
    end
    
    equipWeapon("Katana")
end

-- ========== Shutdown Script ==========
local function shutdownScript()
    pcall(function()
        sendChatMessage("Casca...Casca", "All")
    end)
    
    -- Stop looping animation
    pcall(function()
        if character.loop_animation_track then
            character.loop_animation_track:Stop()
            character.loop_animation_track = nil
        end
        character.loop_animation_playing = false
    end)
    
    -- Restore own character
    pcall(function()
        local char = getCharacter()
        if char then
            for _, part in ipairs(char:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.Transparency = 0
                    part.CanCollide = true
                end
            end
            local rootPart = char:FindFirstChild("HumanoidRootPart")
            if rootPart then
                rootPart.Transparency = 0
                rootPart.CanCollide = true
            end
        end
    end)
    
    -- Restore other players' hitboxes
    pcall(function()
        for _, p in ipairs(players:GetPlayers()) do
            if p ~= player and p.Character then
                local root = p.Character:FindFirstChild("HumanoidRootPart")
                if root then
                    root.Size = Vector3.new(2, 2, 1)
                    root.Transparency = 0
                    root.CanCollide = true
                    root.Material = Enum.Material.Plastic
                    root.BrickColor = BrickColor.new("Medium stone grey")
                end
                for _, part in ipairs(p.Character:GetDescendants()) do
                    if part:IsA("BasePart") and part ~= root then
                        part.Transparency = 0
                        part.CanCollide = true
                    end
                end
            end
        end
    end)
    
    -- Disable immunity
    pcall(function()
        if character.immunity then
            if immunityMaintainConnection then
                immunityMaintainConnection:Disconnect()
                immunityMaintainConnection = nil
            end
            character.immunity = false
        end
    end)
    
    -- Disable floating
    pcall(function()
        if character.floating and character.floatBody then
            character.floatBody:Destroy()
            character.floatBody = nil
            character.floating = false
        end
    end)
    
    -- Disable spin
    pcall(function()
        if character.spin_enabled then
            if character.spin_angularVelocity then
                character.spin_angularVelocity:Destroy()
                character.spin_angularVelocity = nil
            end
            character.spin_enabled = false
            if character.humanoid then
                character.humanoid.AutoRotate = true
            end
        end
    end)
    
    -- Disable hitbox
    pcall(function()
        if character.hitbox_enabled then
            if character.hitbox_connection then
                character.hitbox_connection:Disconnect()
                character.hitbox_connection = nil
            end
            character.hitbox_enabled = false
        end
    end)
    
    -- Disable infinite jump
    pcall(function()
        if character.infinite_jump then
            character.infinite_jump = false
            if jumpConnection then
                jumpConnection:Disconnect()
                jumpConnection = nil
            end
        end
    end)
    
    -- Disable block loop
    pcall(function()
        if character.block_enabled then
            character.block_enabled = false
            if blockLoop then task.cancel(blockLoop); blockLoop = nil end
        end
    end)
    
    -- Disable M1 loop

            character.m1_enabled = false
            if m1Loop then task.cancel(m1Loop); m1Loop = nil end
        end
    end)

    if not character.prediction_enabled then return end
    if not character.follow_target then return end
    
    local targetPlayer = findPlayer(character.follow_target)
    if not targetPlayer or not targetPlayer.Character then return end
    
    local targetRoot = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not targetRoot then return end
    
    local currentPos = targetRoot.Position
    local currentTime = tick()
    
    if character.target_last_pos and character.target_last_time then
        local deltaTime = currentTime - character.target_last_time
        if deltaTime > 0 and deltaTime < 0.5 then
            local velocity = (currentPos - character.target_last_pos) / deltaTime
            if velocity.Magnitude < 500 then
                character.target_velocity = velocity
            end
        end
    end
    
    character.target_last_pos = currentPos
    character.target_last_time = currentTime
end

local function getPredictedPosition(predictionTime)
    if not character.prediction_enabled then return nil end
    
    local targetPlayer = findPlayer(character.follow_target)
    if not targetPlayer or not targetPlayer.Character then return nil end
    
    local targetRoot = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
    if not targetRoot then return nil end
    
    local currentPos = targetRoot.Position
    local velocity = character.target_velocity
    
    if velocity.Magnitude < 1 then return currentPos end
    
    local predictedPos = currentPos + velocity * predictionTime
    local maxDistance = 50
    if (predictedPos - currentPos).Magnitude > maxDistance then
        predictedPos = currentPos + (velocity.Unit * maxDistance)
    end
    
    return predictedPos
end

end

    for _, part in ipairs(parts) do

        character.wallHackActive = false

-- ========== Hitbox Control (Stable – includes shield immunity check) ==========
local hitboxConnection = nil

local function updateHitboxes()
    if not character.hitbox_enabled then return end
    for _, p in ipairs(players:GetPlayers()) do
        if p ~= player then
            -- ★New: check if in slaughter list, if yes skip (immune to area damage)
            if slaughterTargets[p.Name] then
                -- Shielded player, restore their hitbox to normal size
                local char = p.Character
                if char then
                    local root = char:FindFirstChild("HumanoidRootPart")
                    if root then
                        root.Size = Vector3.new(2, 2, 1)  -- restore normal size
                        root.Transparency = 0
                        root.CanCollide = true
                        root.Material = Enum.Material.Plastic
                        root.BrickColor = BrickColor.new("Medium stone grey")
                    end
                end
            else
                if character.hitbox_teamCheck and player.Team == p.Team then
                    -- skip teammates
                else
                    local char = p.Character
                    if char then
                        local root = char:FindFirstChild("HumanoidRootPart")
                        if root then
                            root.Size = Vector3.new(character.hitbox_size, character.hitbox_size, character.hitbox_size)
                            root.Transparency = character.hitbox_transparency
                            root.BrickColor = character.hitbox_color
                            root.Material = Enum.Material.Neon
                            root.CanCollide = false
                        end
                    end
                end
            end
        end
    end
end

    end
    updateUI()
end

-- ========== Damage Immunity ==========
local function setDamageImmunityForPlayer(targetName, enable)
    local targetPlayer = findPlayer(targetName)
    if not targetPlayer then output("Player "..targetName.." is offline", false) return end
    if enable then
        if damageImmunityList[targetPlayer.Name] then task.cancel(damageImmunityList[targetPlayer.Name]); damageImmunityList[targetPlayer.Name] = nil end
        local thread = task.spawn(function()
            while true do
                if not character or not damageImmunityList[targetPlayer.Name] then break end
                local p = players:FindFirstChild(targetPlayer.Name)
                if p and p.Character and p.Character:FindFirstChildOfClass("Humanoid") then
                    local hum = p.Character:FindFirstChildOfClass("Humanoid")
                    if hum and hum.Health < hum.MaxHealth then
                        hum.Health = hum.MaxHealth
                    end
                else
                    output("Player "..targetPlayer.DisplayName.." disappeared, immunity auto-disabled", true)
                    damageImmunityList[targetPlayer.Name] = nil
                    break
                end
                task.wait(0.1)
            end
        end)
        damageImmunityList[targetPlayer.Name] = thread
        output("Damage immunity ON for "..targetPlayer.DisplayName, true)
    else
        if damageImmunityList[targetPlayer.Name] then task.cancel(damageImmunityList[targetPlayer.Name]); damageImmunityList[targetPlayer.Name] = nil end
        output("Damage immunity OFF for "..targetPlayer.DisplayName, true)
    end
    updateUI()
end

-- ========== Reset Character ==========
local function resetCharacter()
    updateCharacterRefs()
    local success, err = pcall(function()
        if character.rootPart then
            character.rootPart.CanCollide = true
            character.rootPart.Transparency = 0
        end
        if character.humanoid then character.humanoid.WalkSpeed = 16 end
        if character.floating then setFloating(false) end
        if character.follow_target then stopFollow() end
        if character.immunity then
            task.wait(0.2)
            applyImmunityToAllParts()
        end
        if character.spin_enabled then toggleSpin(false) end
        if character.hitbox_enabled then toggleHitbox(false) end
        if character.infinite_jump then toggleInfiniteJump() end
        if character.block_enabled then toggleBlock(false) end
        if character.m1_enabled then toggleM1Loop(false) end
        if character.m2_enabled then toggleM2Loop(false) end
        if character.prediction_enabled then togglePrediction(false) end
        if character.aura_farm_enabled then toggleAuraFarm(false) end
        if character.anim_speed_enabled then toggleAnimSpeed(false) end
        character.loop_teleport_to = false
        character.loop_teleport_from = false
        character.loop_suck_all = false
        character.loop_fling = false
        character.aimbot = false
        character.viewing = false
        character.auto_suicide = false
        character.kuangkai_enabled = false
        if character.kuangkai_thread then task.cancel(character.kuangkai_thread); character.kuangkai_thread = nil end
        
        if character.loop_animation_track then
            character.loop_animation_track:Stop()
            character.loop_animation_track = nil
        end
        character.loop_animation_playing = false
        
        for name, thread in pairs(damageImmunityList) do
            task.cancel(thread)
            damageImmunityList[name] = nil
        end
        hitboxImmunePlayers = {}
    end)
    if not success then output("Reset failed: " .. tostring(err), false) return end
    output("Character reset", true)
    updateUI()
end

-- ========== Get player ==========
local function getNearestPlayerName()
    updateCharacterRefs()
    if not character.rootPart then return nil end
    local nearest, minDist = nil, math.huge
    for _, p in ipairs(players:GetPlayers()) do
        if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local dist = (p.Character.HumanoidRootPart.Position - character.rootPart.Position).Magnitude
            if dist < minDist then minDist = dist; nearest = p end
        end
    end
    return nearest
end

local function smoothMoveTo(targetPosition)
    if not character or not character.rootPart then return end
    local currentPos = character.rootPart.Position
    local distance = (targetPosition - currentPos).Magnitude
    if distance < 2 then
        character.rootPart.CFrame = CFrame.new(targetPosition)
        character.rootPart.Velocity = Vector3.new(0,0,0)
        return
    end
    local direction = (targetPosition - currentPos).Unit
    local moveSpeed = math.min(character.speed * 5, 250)
    character.rootPart.Velocity = direction * moveSpeed
    if math.abs(targetPosition.Y - currentPos.Y) > 0.5 then
        setFloating(true)
    end
end

local function smoothMoveHorizontal(targetPosition)
    if not character or not character.rootPart then return end
    local currentPos = character.rootPart.Position
    local targetXZ = Vector3.new(targetPosition.X, currentPos.Y, targetPosition.Z)
    local distance = (targetXZ - currentPos).Magnitude
    if distance < 2 then
        character.rootPart.CFrame = CFrame.new(targetXZ)
        character.rootPart.Velocity = Vector3.new(0,0,0)
        return
    end
    local direction = (targetXZ - currentPos).Unit
    local moveSpeed = math.min(character.speed * 5, 250)
    character.rootPart.Velocity = direction * moveSpeed
end

local function teleportToPlayer(targetPlayer, smooth)
    if not targetPlayer or not targetPlayer.Character or not targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
        output("Player "..getDisplayName(targetPlayer).." is offline or has no character", false)
        return
    end
    updateCharacterRefs()
    if not character.rootPart then
        output("Cannot get your character position", false)
        return
    end
    local targetPart = targetPlayer.Character.HumanoidRootPart
    local targetPos = targetPart.Position
    local dest = Vector3.new(targetPos.X, targetPos.Y, targetPos.Z)
    if smooth then
        smoothMoveTo(dest)
        output("Smooth moving to "..targetPlayer.DisplayName, true)
    else
        character.rootPart.CFrame = CFrame.new(dest)
        character.rootPart.Velocity = Vector3.new(0,0,0)
        if not character.floating then setFloating(true) end
        output("Teleported to "..targetPlayer.DisplayName.."'s position", true)
    end
    if ui.tpInput and ui.tpInput.Parent then
        ui.tpInput.Text = ""
    end
end

end

local function flingTargetOnce()
    local target = getTargetPlayer()
    if not target then
        notify("Pippin", "Please select a target first", 3)
        return
    end
    local localChar = getCharacter()
    if not localChar then return end
    local localHum = localChar:FindFirstChildOfClass("Humanoid")
    local localRoot = localHum and localHum.RootPart
    if not localRoot then return end

    local targetChar = target.Character
    if not targetChar then
        notify("Pippin", "Target character does not exist", 3)
        return
    end
    local targetHum = targetChar:FindFirstChildOfClass("Humanoid")
    local targetRoot = targetHum and targetHum.RootPart
    local targetHead = targetChar:FindFirstChild("Head")
    if not targetRoot and not targetHead then
        notify("Pippin", "Target has no flingable part", 3)
        return
    end

    getgenv().OldPos = localRoot.CFrame
    getgenv().FPDH = workspace.FallenPartsDestroyHeight
    workspace.FallenPartsDestroyHeight = 0/0

    if targetHead then
        workspace.CurrentCamera.CameraSubject = targetHead
    elseif targetRoot then
        workspace.CurrentCamera.CameraSubject = targetHum
    end

    localHum:SetStateEnabled(Enum.HumanoidStateType.Seated, false)

    local bv = Instance.new("BodyVelocity")
    bv.Name = "FlingVelocity"
    bv.Velocity = Vector3.new(9e7, 9e8, 9e7)
    bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    bv.Parent = localRoot

    local targetPart = targetHead or targetRoot
    local startTime = tick()
    local timeout = 3

    while localRoot and targetPart and targetPart.Parent do
        local dt = tick() - startTime
        if dt > timeout then break end
        local angle = dt * 500
        local offset = CFrame.new(0, 1.5, 0) * CFrame.Angles(0, math.rad(angle), 0)
        localRoot.CFrame = CFrame.new(targetPart.Position) * offset
        localChar:SetPrimaryPartCFrame(CFrame.new(targetPart.Position) * offset)
        task.wait()
    end

    bv:Destroy()
    localHum:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
    workspace.CurrentCamera.CameraSubject = localHum

    repeat
        localRoot.CFrame = getgenv().OldPos * CFrame.new(0, 0.5, 0)
        localChar:SetPrimaryPartCFrame(getgenv().OldPos * CFrame.new(0, 0.5, 0))
        localHum:ChangeState("GettingUp")
        for _, part in ipairs(localChar:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Velocity = Vector3.new()
                part.RotVelocity = Vector3.new()
            end
        end
        task.wait()
    until (localRoot.Position - getgenv().OldPos.p).Magnitude < 25

    workspace.FallenPartsDestroyHeight = getgenv().FPDH
    notify("Pippin", "Fling complete", 3)
end

-- ========== Dodge (kuangkai) ==========
local function toggleKuangkai(state)
    if state == nil then
        character.kuangkai_enabled = not character.kuangkai_enabled
    else
        character.kuangkai_enabled = state
    end

    if character.kuangkai_enabled then
        if character.kuangkai_thread then task.cancel(character.kuangkai_thread) end

        if not character.follow_target then
            output("Please set a hunt target first (.hunt <PlayerName>)", false)
            character.kuangkai_enabled = false
            return
        end

        character.kuangkai_thread = task.spawn(function()
            local angle = 0
            local distance = 181
            local speed = 0.4
            
            while character.kuangkai_enabled do
                local targetName = character.follow_target
                if not targetName then
                    output("Target lost, stopping dodge", true)
                    character.kuangkai_enabled = false
                    break
                end

                local targetPlayer = findPlayer(targetName)
                if not targetPlayer or not targetPlayer.Character then
                    output("Target character doesn't exist, stopping dodge", true)
                    character.kuangkai_enabled = false
                    break
                end

                local targetHum = targetPlayer.Character:FindFirstChildOfClass("Humanoid")
                local targetPart = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                
                if not targetHum or targetHum.Health <= 0 or not targetPart then
                    output("Target is dead or invalid, stopping dodge", true)
                    character.kuangkai_enabled = false
                    break
                end

                updateCharacterRefs()
                if character.rootPart then
                    local targetPos = targetPart.Position

                    local offsetX = math.cos(angle) * distance
                    local offsetZ = math.sin(angle) * distance
                    local newPos = Vector3.new(
                        targetPos.X + offsetX,
                        targetPos.Y + 2,
                        targetPos.Z + offsetZ
                    )

                    character.rootPart.CFrame = CFrame.lookAt(newPos, targetPos)
                    character.rootPart.Velocity = Vector3.new(0,0,0)
                    if not character.floating then setFloating(true) end

                    local cam = workspace.CurrentCamera
                    cam.CameraType = Enum.CameraType.Scriptable
                    local camDistance = 20
                    local camHeight = 8
                    local camPos = targetPos + Vector3.new(math.cos(angle) * camDistance, camHeight, math.sin(angle) * camDistance)
                    cam.CFrame = CFrame.lookAt(camPos, targetPos)

                    angle = angle + speed

                    if angle % (math.pi) < speed then
                        distance = (distance == 14) and 20 or 14
                    end

                    if angle > 2 * math.pi then
                        angle = angle - 2 * math.pi
                    end
                end
                
                task.wait(0.05)
            end

            updateCharacterRefs()
            if character.humanoid then
                workspace.CurrentCamera.CameraSubject = character.humanoid
                workspace.CurrentCamera.CameraType = Enum.CameraType.Custom
            end
        end)
        output("Dodge ON: God perspective circling target", true)
    else
        if character.kuangkai_thread then
            task.cancel(character.kuangkai_thread)
            character.kuangkai_thread = nil
        end
        updateCharacterRefs()
        if character.humanoid then
            workspace.CurrentCamera.CameraSubject = character.humanoid
            workspace.CurrentCamera.CameraType = Enum.CameraType.Custom
        end
        output("Dodge OFF", true)
    end
    updateUI()
end

-- ========== Hunting ==========
local function stopFollow()
    if character.follow_target then
        character.follow_target = nil
        if followLoop then task.cancel(followLoop); followLoop = nil end
        output("Hunting stopped", true)
        updateUI()
    end
end

local function startFollow(targetName)
    if character.follow_target then stopFollow() end
    local targetPlayer = findPlayer(targetName)
    if not targetPlayer then output("Player "..targetName.." is offline, but will keep waiting...", true) end
    character.follow_target = targetName
    if followLoop then task.cancel(followLoop); followLoop = nil end
    followLoop = task.spawn(function()
        local waitTimer = 0
        while character and character.follow_target do
            local target = findPlayer(character.follow_target)
            if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
                updateCharacterRefs()
                if character and character.rootPart then
                    local targetChar = target.Character
                    local targetPart = targetChar:FindFirstChild("HumanoidRootPart")
                    local targetHead = targetChar:FindFirstChild("Head")
                    local targetBaseY = targetHead and targetHead.Position.Y or (targetPart.Position.Y + 2.5)
                    local targetY = targetBaseY + character.follow_height_offset
                    local targetPos = targetPart.Position
                    
                    if character.prediction_enabled then
                        local predictedPos = getPredictedPosition(0.1)
                        if predictedPos then
                            targetPos = predictedPos
                        end
                    end

                    if character.smooth_follow then
                        local currentPos = character.rootPart.Position
                        character.rootPart.CFrame = CFrame.new(currentPos.X, targetY, currentPos.Z)
                        smoothMoveHorizontal(Vector3.new(targetPos.X, targetY, targetPos.Z))
                    else
                        character.rootPart.CFrame = CFrame.new(targetPos.X, targetY, targetPos.Z)
                        character.rootPart.Velocity = Vector3.new(0,0,0)
                        if not character.floating then setFloating(true) end
                    end
                end
                waitTimer = 0
                task.wait(character.follow_interval)
            else
                waitTimer = waitTimer + 1
                if waitTimer > 30 then
                    output("Hunt target went offline, auto-stopping", true)
                    character.follow_target = nil
                    break
                end
                task.wait(0.1 * waitTimer)
            end
        end
        updateUI()
    end)
    if targetPlayer then
        output("Hunting "..targetPlayer.DisplayName.." (head offset +"..character.follow_height_offset..")", true)
        sendChatMessage("👹Hunting " .. targetPlayer.DisplayName .. "!!!", "All")
    end
    updateUI()
end

-- ========== Command Parsing (fixed huntlist invalid & unshield health restore, added ambush control) ==========
local function executeCommand(cmd)
    cmd = cmd:gsub("^%s+", ""):gsub("%s+$", "")
    -- Handle double-dot prefix, e.g. ..huntlist -> .huntlist
    if cmd:sub(1,2) == ".." then
        cmd = "." .. cmd:sub(3)
    end

    if not bot_active then
        if PASSWORD == nil then return end
        if cmd == PASSWORD then
            bot_active = true
            output("Pippin activated! Type .help for all commands", true)
            
            local messages = {
                "[☄️Eclipse☄️] Steady Pippin, made by etern_ncz(hidden) [☄️Eclipse☄️]",
                ".hitbox",
                ".hitbox",
                ".m1",
                ".m2",
                ".b",
                ".j"
            }
            task.spawn(function()
                for _, msg in ipairs(messages) do
                    sendChatMessage(msg, "All")
                    task.wait(2)
                end
            end)
            
            setImmunity(true)
            toggleInfiniteJump()
            character.hitbox_size = 1200
            toggleHitbox(true)
            output("⚡ Auto-enabled: No-clip, Infinite Jump, Hitbox(1200)", true)
            output("⚡ .autohunt is OFF by default, type .autohunt to enable", true)
            output("⚡ .ambush toggles server hop permission (current: " .. (allowServerHop and "Allow" or "Deny") .. ")", true)
            
            task.spawn(function()
                task.wait(3)
                autoEquipWeapon()
            end)
            
            updateUI()
        end
        return
    end
    
    if cmd == "" then return end

    if cmd == "严阵以待！！！" or cmd == "Stand by!!!" or cmd == "Stand by" then
        if not character.stand_by_message_sent then
            sendChatMessage("Kill!!!", "All")
            character.stand_by_message_sent = true
        end
        
        local char = getCharacter()
        if char then
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then
                local animator = hum:FindFirstChild("Animator")
                if animator then
                    if character.loop_animation_playing then
                        if character.loop_animation_track then
                            character.loop_animation_track:Stop()
                            character.loop_animation_track = nil
                        end
                        character.loop_animation_playing = false
                    end
                    
                    character.loop_animation_playing = true
                    
                    task.spawn(function()
                        local animationId = "rbxassetid://89900565430002"
                        while character.loop_animation_playing and hum and hum.Parent do
                            local animation = Instance.new("Animation")
                            animation.AnimationId = animationId
                            local animationTrack = animator:LoadAnimation(animation)
                            character.loop_animation_track = animationTrack
                            animationTrack:Play()
                            
                            local animLength = animationTrack.Length
                            if animLength > 0 then
                                task.wait(animLength)
                            else
                                task.wait(1)
                            end
                            
                            if not character.loop_animation_playing then
                                break
                            end
                        end
                    end)
                    output("🔁 Started loop animation", true)
                else
                    output("Cannot get Animator", false)
                end
            else
                output("Cannot get Humanoid", false)
            end
        else
            output("Cannot get character", false)
        end
        return
    end

    if cmd:sub(1,1) == "." then
        local dotcmd, arg = cmd:match("^%.([^%s]+)%s*(.*)$")
        if not dotcmd then
            dotcmd = cmd:match("^%.([^%s]+)$")
            arg = ""
        end
        if not dotcmd then return end
        dotcmd = dotcmd:lower()
        
        local cmdMap = {
            ["帮助"] = "help", ["help"] = "help",
            ["穿墙"] = "w", ["悬浮"] = "fl", ["fly"] = "fl",
            ["速度+"] = "s", ["重置"] = "r", ["自杀"] = "suicide",
            ["格挡"] = "b", ["m1"] = "m1", ["m2"] = "m2", ["组合"] = "c",
            ["狩猎"] = "follow", ["狩猎停"] = "followstop",
            ["传送"] = "tp", ["拉回"] = "pull", ["吸全部"] = "suck",
            ["无限跳"] = "j", ["旋转"] = "spin", ["范围"] = "hitbox",
            ["自瞄"] = "aim", ["查看"] = "view", ["甩飞"] = "fling",
            [".k"] = "kuangkai", ["预判"] = "pred",
            ["授权"] = "ac", ["取消授权"] = "rm", ["授权列表"] = "authlist",
            ["高度"] = "y", ["y锁"] = "ylock", ["头顶"] = "fh",
            ["玩家列表"] = "p", ["防afk"] = "afk", ["防AFK"] = "afk",
            ["aurafarm"] = "afarm", ["af"] = "afarm", ["动画"] = "anim", ["anim"] = "anim",
            ["关闭"] = "shutdown", ["shutdown"] = "shutdown",
            ["彻底关闭"] = "shutdown", ["完全关闭"] = "shutdown", ["关闭脚本"] = "shutdown",
            ["停止动画"] = "stopanim", ["stopanim"] = "stopanim",
            ["shield"] = "庇护", ["取消庇护"] = "unshield",
            ["名单"] = "huntlist", ["huntlist"] = "huntlist",
            ["追魂夺命"] = "huntlist",   -- ★ ensure .追魂夺命 goes to hunt list management
            ["自动狩猎"] = "autohunt", ["autohunt"] = "autohunt",
            ["重叠"] = "overlap", ["overlap"] = "overlap",
            ["伏击"] = "ambush", ["伏击狩猎"] = "ambush", ["ambush"] = "ambush",
            ["允许换服"] = "allowhop", ["allowhop"] = "allowhop",
            ["禁止换服"] = "denyhop", ["denyhop"] = "denyhop",
        }
        
        local mapped = cmdMap[dotcmd]
        if mapped then
            dotcmd = mapped
        end

        if dotcmd == "help" then
            showHelp()
        elseif dotcmd == "pred" then
            togglePrediction()
        elseif dotcmd == "afk" then
            character.antiAFK = not character.antiAFK
            output("AntiAFK " .. (character.antiAFK and "ON" or "OFF"), true)
            updateUI()
        elseif dotcmd == "w" then
            setImmunity(not character.immunity)
        elseif dotcmd == "j" then
            toggleInfiniteJump()
        elseif dotcmd == "s" then
            adjustSpeed()
        elseif dotcmd == "r" then
            resetCharacter()
        elseif dotcmd == "suicide" then
            local char = getCharacter()
            if char then
                local hum = char:FindFirstChildOfClass("Humanoid")
                if hum then hum.Health = 0 end
                char:BreakJoints()
            end
            output("Suicide executed", true)
        elseif dotcmd == "p" then
            refreshPlayerList()
        elseif dotcmd == "ambush" then
            allowServerHop = not allowServerHop
            if allowServerHop then
                output("Server hop permission ON: will auto hop when no target in list", true)
            else
                output("Server hop permission OFF: will not hop (Ambush mode)", true)
            end
            updateUI()
        elseif dotcmd == "allowhop" then
            allowServerHop = true
            output("Server hop forced ON", true)
            updateUI()
        elseif dotcmd == "denyhop" then
            allowServerHop = false
            output("Server hop forced OFF (Ambush mode)", true)
            updateUI()
        elseif dotcmd == "huntlist" or dotcmd == "名单" then
            -- ★ Fixed huntlist add/remove/list logic
            -- Prioritize stop commands
            if arg:match("^%s*停止$") or arg:match("^%s*stop$") or arg:match("^%s*关闭$") then
                if autoHuntEnabled then
                    autoHuntEnabled = false
                    output("⚡ Auto-hunt stopped", true)
                    updateUI()
                else
                    output("Auto-hunt is already stopped", true)
                end
                return
            end

            local action, targetName = nil, nil
            if arg:match("^%s*添加%s+") or arg:match("^%s*add%s+") then
                action = "add"
                targetName = arg:match("添加%s+(.+)") or arg:match("add%s+(.+)")
            elseif arg:match("^%s*删除%s+") or arg:match("^%s*remove%s+") then
                action = "remove"
                targetName = arg:match("删除%s+(.+)") or arg:match("remove%s+(.+)")
            elseif arg:match("^%s*列表$") or arg:match("^%s*list$") or arg == "" then
                action = "list"
            end

            if action == "add" then
                if targetName and targetName ~= "" then
                    local target = findPlayer(targetName)
                    if target then
                        huntList[target.Name] = true
                        saveHuntList()
                        output("✅ Added " .. target.DisplayName .. " to hunt list", true)
                    else
                        huntList[targetName] = true
                        saveHuntList()
                        output("⚠️ Player not online, added name " .. targetName .. " to list (verify spelling)", true)
                    end
                else
                    output("❌ Usage: .huntlist add <PlayerName>", false)
                end
            elseif action == "remove" then
                if targetName and targetName ~= "" then
                    local target = findPlayer(targetName)
                    local removed = false
                    if target and huntList[target.Name] then
                        huntList[target.Name] = nil
                        removed = true
                        output("✅ Removed " .. target.DisplayName .. " from list", true)
                    else
                        for storedName in pairs(huntList) do
                            if storedName:lower() == targetName:lower() then
                                huntList[storedName] = nil
                                removed = true
                                output("✅ Removed " .. storedName .. " from list", true)
                                break
                            end
                        end
                    end
                    if not removed then
                        output("❌ Player not found in list: " .. targetName, false)
                    else
                        saveHuntList()
                    end
                else
                    output("❌ Usage: .huntlist remove <PlayerName>", false)
                end
            elseif action == "list" then
                local names = {}
                for name in pairs(huntList) do
                    local p = players:FindFirstChild(name)
                    table.insert(names, p and p.DisplayName or name.."(offline)")
                end
                if #names > 0 then
                    output("📋 Hunt list: " .. table.concat(names, ", "), true)
                else
                    output("📋 Hunt list is empty", true)
                end
            else
                output("❌ Usage: .huntlist add/remove/list/stop", false)
            end
        elseif dotcmd == "autohunt" or dotcmd == "自动狩猎" then
            autoHuntEnabled = not autoHuntEnabled
            if autoHuntEnabled then
                lastRejoinTime = tick()  -- reset cooldown to prevent immediate hop
                output("⚡ Auto-hunt ON - will track list players, no target will hop after 1 min (hop permission: " .. (allowServerHop and "Allow" or "Deny") .. ")", true)
                output("⚠️ Warning: If no list players online and hop allowed, will auto server hop!", true)
            else
                output("⚡ Auto-hunt OFF - stopped auto tracking and hopping", true)
            end
            updateUI()
        elseif dotcmd == "overlap" or dotcmd == "重叠" then
            character.ghostModeEnabled = not character.ghostModeEnabled
            if character.ghostModeEnabled then
                local myChar = getCharacter()
                if myChar then
                    for _, part in ipairs(myChar:GetDescendants()) do
                        if part:IsA("BasePart") then
                            part.CanCollide = false
                            part.Transparency = 0.3
                        end
                    end
                end
                output("Overlap mode ON (mutual immunity with list players)", true)
            else
                local myChar = getCharacter()
                if myChar then
                    for _, part in ipairs(myChar:GetDescendants()) do
                        if part:IsA("BasePart") then
                            part.CanCollide = true
                            part.Transparency = 0
                        end
                    end
                end
                output("Overlap mode OFF", true)
            end
            updateUI()
        elseif dotcmd == "庇护" then
            if arg and arg ~= "" then
                local targetPlayer = findPlayer(arg)
                if targetPlayer then
                    if slaughterTargets[targetPlayer.Name] then
                        output(targetPlayer.DisplayName .. " is already shielded", false)
                    else
                        slaughterTargets[targetPlayer.Name] = true
                        output("🛡️ Shielded " .. targetPlayer.DisplayName .. " (HP locked 0, immune to hitbox)", true)
                        sendChatMessage("🛡️ Shield " .. targetPlayer.DisplayName .. "!!!", "All")
                        startSlaughter()
                    end
                else
                    output("Player not found: " .. arg, false)
                    refreshPlayerList()
                end
            else
                output("Usage: .shield <PlayerName/Num/Short>", false)
            end
        elseif dotcmd == "unshield" or dotcmd == "取消庇护" then
            if arg and arg ~= "" then
                local targetPlayer = findPlayer(arg)
                local removedKey = nil
                if targetPlayer then
                    if slaughterTargets[targetPlayer.Name] then
                        slaughterTargets[targetPlayer.Name] = nil
                        removedKey = targetPlayer.Name
                        output("Unshielded " .. targetPlayer.DisplayName, true)
                        sendChatMessage(targetPlayer.DisplayName .. " shield removed", "All")
                    else
                        for name in pairs(slaughterTargets) do
                            if name:lower() == targetPlayer.Name:lower() then
                                slaughterTargets[name] = nil
                                removedKey = name
                                output("Unshielded " .. name, true)
                                break
                            end
                        end
                    end
                    if removedKey then
                        restorePlayerHealth(removedKey)  -- ★ Restore health
                    end
                else
                    for name in pairs(slaughterTargets) do
                        local p = players:FindFirstChild(name)
                        local displayName = p and p.DisplayName or name
                        if displayName:lower():find(arg:lower(), 1, true) then
                            slaughterTargets[name] = nil
                            output("Unshielded " .. displayName, true)
                            restorePlayerHealth(name)
                            removedKey = name
                            break
                        end
                    end
                    if not removedKey then
                        output("Shielded player not found: " .. arg, false)
                    end
                end
            else
                local names = {}
                for name in pairs(slaughterTargets) do
                    local p = players:FindFirstChild(name)
                    table.insert(names, p and p.DisplayName or name.."(offline)")
                end
                if #names > 0 then
                    output("Current shielded: " .. table.concat(names, ", "), true)
                else
                    output("No players are shielded", true)
                end
            end
        elseif dotcmd == "follow" then
            if arg and arg ~= "" then
                if arg == "stop" or arg == "停" then
                    stopFollow()
                else
                    local target = findPlayer(arg)
                    if target then
                        startFollow(target.DisplayName)
                    else
                        output("Player not found: " .. arg, false)
                        refreshPlayerList()
                    end
                end
            else
                followNearest()
            end
        elseif dotcmd == "followstop" then
            stopFollow()
        elseif dotcmd == "tp" then
            if arg and arg ~= "" then
                local target = findPlayer(arg)
                if target then
                    teleportToPlayer(target, character.smooth_follow)
                else
                    output("Player not found: " .. arg, false)
                    refreshPlayerList()
                end
            else
                local nearest = getNearestPlayerName()
                if nearest then teleportToPlayer(nearest, character.smooth_follow) end
            end
        elseif dotcmd == "pull" then
            if arg and arg ~= "" then
                local target = findPlayer(arg)
                if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
                    updateCharacterRefs()
                    if character.rootPart then
                        local myPos = character.rootPart.Position
                        target.Character.HumanoidRootPart.CFrame = CFrame.new(myPos + Vector3.new(0, 0, 3))
                        target.Character.HumanoidRootPart.Velocity = Vector3.new(0,0,0)
                        output("Pulled "..target.DisplayName.." to you", true)
                    end
                else
                    output("Player not found: " .. arg, false)
                end
            else
                output("Usage: .pull <PlayerName>", false)
            end
        elseif dotcmd == "suck" then
            toggleSuckAll(not character.loop_suck_all)
        elseif dotcmd == "b" then
            toggleBlock()
        elseif dotcmd == "m1" then
            toggleM1Loop()
        elseif dotcmd == "m2" then
            toggleM2Loop()
        elseif dotcmd == "afarm" then
            toggleAuraFarm()
        elseif dotcmd == "anim" then
            if arg and arg ~= "" then
                setAnimSpeedMultiplier(arg)
            else
                toggleAnimSpeed()
            end
        elseif dotcmd == "c" then
            local any_on = character.block_enabled or character.m1_enabled or character.m2_enabled or character.aura_farm_enabled or character.anim_speed_enabled
            if any_on then
                if character.block_enabled then toggleBlock(false) end
                if character.m1_enabled then toggleM1Loop(false) end
                if character.m2_enabled then toggleM2Loop(false) end
                if character.aura_farm_enabled then toggleAuraFarm(false) end
                if character.anim_speed_enabled then toggleAnimSpeed(false) end
                output("All loops turned OFF", true)
            else
                if not character.block_enabled then toggleBlock(true) end
                if not character.m1_enabled then toggleM1Loop(true) end
                if not character.m2_enabled then toggleM2Loop(true) end
                if not character.aura_farm_enabled then toggleAuraFarm(true) end
                if not character.anim_speed_enabled then toggleAnimSpeed(true) end
                output("All loops turned ON", true)
            end
        elseif dotcmd == "as" then
            toggleAutoSuicide()
        elseif dotcmd == "ac" then
            if arg ~= "" then
                local targetPlayer = findPlayer(arg)
                if targetPlayer then
                    authorizePlayer(targetPlayer)
                    output("✅ Authorized player " .. targetPlayer.DisplayName .. " for remote control", true)
                    sendChatMessageToPlayer(targetPlayer, "You have been authorized to remotely control Pippin!")
                    sendChatMessage("Player " .. targetPlayer.DisplayName .. " is now Pippin operator", "All")
                else
                    output("❌ Player not found: " .. arg, false)
                    refreshPlayerList()
                end
            else
                output("Usage: .auth <PlayerName/Num/Short>", false)
            end
        elseif dotcmd == "rm" then
            if arg ~= "" then
                local targetPlayer = findPlayer(arg)
                if targetPlayer then
                    if remoteCommanders[targetPlayer.Name] then
                        deauthorizePlayer(targetPlayer)
                        output("✅ Deauthorized player " .. targetPlayer.DisplayName, true)
                        sendChatMessageToPlayer(targetPlayer, "Your Pippin remote control has been revoked")
                    else
                        output("Player " .. targetPlayer.DisplayName .. " is not authorized", false)
                    end
                else
                    output("❌ Player not found: " .. arg, false)
                    refreshPlayerList()
                end
            else
                output("Usage: .deauth <PlayerName/Num/Short>", false)
            end
        elseif dotcmd == "authlist" then
            local list = {}
            for name, _ in pairs(remoteCommanders) do
                local p = players:FindFirstChild(name)
                if p then
                    list[#list+1] = p.DisplayName
                else
                    list[#list+1] = name .. "(offline)"
                end
            end
            if #list > 0 then
                output("Authorized players: " .. table.concat(list, ", "), true)
            else
                output("No authorized players", true)
            end
        elseif dotcmd == "bcd" then
            if arg and arg ~= "" then
                setBlockCooldown(arg)
            else
                output("Usage: .bcd <seconds>", false)
            end
        elseif dotcmd == "fh" then
            if arg and arg ~= "" then
                setFollowHeightOffset(arg)
            else
                output("Usage: .headoffset <value>", false)
            end
        elseif dotcmd == "yl" then
            toggleYLock()
        elseif dotcmd == "ylock" then
            if arg and arg ~= "" then
                setYLockValue(arg)
            else
                output("Usage: .ylock <height>", false)
            end
        elseif dotcmd == "kuangkai" then
            toggleKuangkai()
        elseif dotcmd == "y" then
            if arg and arg ~= "" then
                setCharacterHeight(arg)
            else
                output("Usage: .height <value>", false)
            end
        elseif dotcmd == "fl" then
            setFloating(not character.floating)
        elseif dotcmd == "mode" then
            toggleFollowMode()
        elseif dotcmd == "spin" then
            if arg == "" then
                toggleSpin()
            elseif arg:match("^%d+$") then
                setSpinSpeed(tonumber(arg))
            else
                output("Usage: .spin or .spin <speed>", false)
            end
        elseif dotcmd == "hitbox" then
            if arg == "" then
                toggleHitbox(not character.hitbox_enabled)
            elseif arg:match("^%d+$") then
                setHitboxSize(tonumber(arg))
            else
                output("Usage: .hitbox or .hitbox <size>", false)
            end
        elseif dotcmd == "teamcheck" then
            toggleTeamCheck()
        elseif dotcmd == "fling" then
            if arg == "" then
                flingTargetOnce()
            else
                output("Usage: .fling (no args, flings current target)", false)
            end
        elseif dotcmd == "view" then
            if arg and arg ~= "" then
                local target = findPlayer(arg)
                if target then
                    character.selected_target = target.DisplayName
                    if ui.targetInput then ui.targetInput.Text = target.DisplayName end
                    toggleViewPlayer(not character.viewing)
                else
                    output("Player not found: " .. arg, false)
                end
            else
                toggleViewPlayer(not character.viewing)
            end
        elseif dotcmd == "aim" then
            toggleAimbot(not character.aimbot)
        elseif dotcmd == "shutdown" then
            shutdownScript()
            return
        elseif dotcmd == "stopanim" then
            if character.loop_animation_playing then
                if character.loop_animation_track then
                    character.loop_animation_track:Stop()
                    character.loop_animation_track = nil
                end
                character.loop_animation_playing = false
                output("⏹️ Loop animation stopped", true)
            else
                output("No loop animation playing", false)
            end
        else 
            output("Unknown command: ."..dotcmd .. " Type .help for commands", false) 
        end
        return
    end

    -- Commands without a dot
    local lcmd = cmd:lower()
    if lcmd == "闲置" or lcmd == "idle" then
        character.attacking = false
        output("Idle state", true)
        updateUI()
    elseif lcmd == "重置" or lcmd == "reset" then
        resetCharacter()
    elseif lcmd == "速度" or lcmd == "speed" then
        adjustSpeed()
    elseif lcmd == "列表" or lcmd == "list" then
        showHelp()
    elseif lcmd == "自动狩猎" or lcmd == "autohunt" then
        autoHuntEnabled = not autoHuntEnabled
        if autoHuntEnabled then
            lastRejoinTime = tick()
            output("⚡ Auto-hunt ON - will track list players, hop after 1 min if none online (hop allowed: " .. (allowServerHop and "Yes" or "No") .. ")", true)
            output("⚠️ Warning: If hop is allowed, will auto rejoin server!", true)
        else
            output("⚡ Auto-hunt OFF", true)
        end
        updateUI()
    else
        output("Unknown command, type .help for help", false)
    end
end

-- ========== Remote Control Listener ==========
local function attachChatListener(p)
    p.Chatted:Connect(function(m)
        if remoteCommanders[p.Name] then
            executeCommand(m)
        end
    end)
end

-- ========== Main Loop ==========
local function run()
    loadHuntList()
    
    if AUTO_ACTIVATE and PASSWORD then
        task.spawn(function()
            task.wait(1)
            executeCommand(PASSWORD)
        end)
    end
    
    if heartbeatConnection then heartbeatConnection:Disconnect(); heartbeatConnection = nil end
    
    showLoadingScreen()
    
    createUI()
    createMenuButton()

    for _, p in ipairs(players:GetPlayers()) do
        if p ~= player then
            attachChatListener(p)
        end
    end

    players.PlayerAdded:Connect(attachChatListener)

    players.PlayerAdded:Connect(function(newPlayer)
        updatePlayerList()
    end)
    players.PlayerRemoving:Connect(function()
        updatePlayerList()
    end)

    player.CharacterAdded:Connect(function()
        task.wait(0.5)
        updateCharacterRefs()
        if character.floatBody then pcall(function() character.floatBody:Destroy() end); character.floatBody = nil end
        character.floating = false
        if character.immunity then
            task.wait(0.2)
            applyImmunityToAllParts()
        end
        if character.spin_enabled then toggleSpin(true) end
        if character.block_enabled then toggleBlock(true) end
        if character.m1_enabled then toggleM1Loop(true) end
        if character.m2_enabled then toggleM2Loop(true) end
        if character.aura_farm_enabled then toggleAuraFarm(true) end
        if character.anim_speed_enabled then toggleAnimSpeed(true) end
        updateUI()
    end)

    local lastAntiAFK = tick()

    heartbeatConnection = runService.Heartbeat:Connect(function()
        if bot_active and character then
            -- Auto-hunt list scan and server hop (only when autoHuntEnabled is ON)
            if autoHuntEnabled then
                local now = tick()
                if not huntScanTimer or now - huntScanTimer >= huntScanInterval then
                    huntScanTimer = now
                    
                    local isHuntingListTarget = false
                    if character.follow_target then
                        local targetPlayer = findPlayer(character.follow_target)
                        if targetPlayer and huntList[targetPlayer.Name] then
                            isHuntingListTarget = true
                        end
                    end
                    
                    if not isHuntingListTarget then
                        local foundPlayer = nil
                        for name, _ in pairs(huntList) do
                            local p = players:FindFirstChild(name)
                            if p and p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                                foundPlayer = p
                                break
                            end
                        end
                        
                        if foundPlayer then
                            startFollow(foundPlayer.DisplayName)
                            output("Auto-hunting list player: " .. foundPlayer.DisplayName, true)
                        else
                            if character.follow_target then
                                stopFollow()
                            end
                            if now - lastRejoinTime > REJOIN_COOLDOWN then
                                if allowServerHop then
                                    output("No list player online, hopping server in 2 seconds...", true)
                                    lastRejoinTime = now
                                    task.wait(2)
                                    pcall(function()
                                        game:GetService("TeleportService"):Teleport(game.PlaceId, player)
                                    end)
                                else
                                    output("No list player online, but hopping is forbidden (Ambush mode), continuing to wait...", true)
                                    lastRejoinTime = now -- update cooldown to avoid spamming
                                end
                            end
                        end
                    end
                end
            end
            
            updateUI()
            maintainSpeed()
            updateTargetVelocity()
            
            if character.anim_speed_enabled then
                applyAnimSpeed()
            end
            
            if character.trigger_animation_id then
                local char = getCharacter()
                if char then
                    local hum = char:FindFirstChildOfClass("Humanoid")
                    if hum then
                        local animator = hum:FindFirstChild("Animator")
                        if animator then
                            for _, track in pairs(animator:GetPlayingAnimationTracks()) do
                                if track.Animation and track.Animation.AnimationId then
                                    local trackId = track.Animation.AnimationId
                                    local targetId = character.trigger_animation_id
                                    
                                    local trackNum = trackId:match("(%d+)")
                                    local targetNum = targetId:match("(%d+)")
                                    
                                    if trackNum and targetNum and trackNum == targetNum then
                                        if character.rootPart then
                                            local currentPos = character.rootPart.Position
                                            local newPos = Vector3.new(currentPos.X, currentPos.Y + 100, currentPos.Z)
                                            
                                            local targetPlayer = character.follow_target and findPlayer(character.follow_target)
                                            if targetPlayer and targetPlayer.Character then
                                                local targetPart = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                                                if targetPart then
                                                    local targetPos = targetPart.Position
                                                    local dirFromTarget = (currentPos - targetPos).Unit
                                                    local farPos = targetPos + dirFromTarget * 100
                                                    newPos = Vector3.new(farPos.X, currentPos.Y + 100, farPos.Z)
                                                    output("⚠️ Detected trigger animation, moved up 100 and away 100 from target", true)
                                                else
                                                    output("⚠️ Detected trigger animation, moved up 100", true)
                                                end
                                            else
                                                output("⚠️ Detected trigger animation, moved up 100", true)
                                            end
                                            
                                            character.rootPart.CFrame = CFrame.new(newPos)
                                            character.rootPart.Velocity = Vector3.new(0,0,0)
                                            if not character.floating then setFloating(true) end
                                        end
                                        break
                                    end
                                end
                            end
                        end
                    end
                end
            end

                end
            end

                end
            end
